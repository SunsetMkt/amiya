# 组合表单

```tsx
import React, { useState, useRef } from 'react'
import { AyForm, AyButton, AyFormField } from 'amiya'
import { Switch, Space } from 'antd'

const fields: Array<AyFormField> = [
  {
    title: '组合输入',
    // group 的 key 可以不写，内部会用 Math.random() 随机一个 key，但是还是建议写一个，以双下划线开头，提交时并不会有这个 key 的数据
    key: '__group',
    // type 为 group，表示子元素为组合表单
    type: 'group',
    required: true,
    children: [
      {
        // 即使写了也不会产生标题，但是会影响 placeholder
        title: 'A',
        key: 'a',
        // type: 'select',
        required: true,
        span: 8,
        options: [{ label: '选项1', value: '1' }]
      },
      // {
      //   // 此处的 key 随意
      //   key: '__divider',
      //   type: 'custom',
      //   // 此处 span 看着来
      //   span: 2,
      //   render: () => {
      //     // 你可以渲染一个分割线，此 class 会带一个默认的样式
      //     return <div className="ay-form-divider">-</div>
      //   }
      // },
      {
        title: 'B',
        key: 'b',
        required: true,
        span: 8
      },
      {
        title: 'date',
        key: 'date',
        type: 'date',
        span: 8,
        required: true
      }
    ]
  },
  {
    title: '组合输入2',
    key: '__group2',
    // 如果想要两个输入框之间的样式毫无间隙，则需要设置为 input-group
    type: 'input-group',
    required: true,
    children: [
      {
        title: 'D',
        key: 'd',
        required: true,
        // 需要手动设置输入框宽度，不然会撑满，此处设置 span 无效
        props: {
          style: {
            width: '33%'
          }
        }
      },
      {
        title: 'E',
        key: 'e',
        required: true,
        // 需要手动设置输入框宽度，不然会撑满，此处设置 span 无效
        props: {
          style: {
            width: '33%'
          }
        }
      },
      {
        title: 'bb',
        key: 'bb',
        required: true,
        type: 'date-range',
        // 需要手动设置输入框宽度，不然会撑满，此处设置 span 无效
        props: {
          style: {
            width: '33%'
          }
        }
      }
    ]
  }
]

export default function Demo() {
  const [readonly, setReadonly] = useState<boolean>(false)
  const [desc, setDesc] = useState<boolean>(false)
  const formRef = useRef()

  const handleConfirm = (form: any) => {
    console.log(form)
    alert(JSON.stringify(form))
  }

  const getFieldsValue = () => {
    let values = formRef.current.getFormatFieldsValue()
    console.log(values)
    alert(JSON.stringify(values))
  }

  const setFieldsValue = () => {
    formRef.current.setFieldsValue({
      a: 11,
      b: 22,
      d: 33,
      e: 44
    })
  }

  return (
    <>
      <p>
        <label style={{ marginRight: 4 }}>只读模式</label>
        <Switch defaultChecked={readonly} onChange={value => setReadonly(value)} />
        <label style={{ marginRight: 4, marginLeft: 10 }}>Desc</label>
        <Switch defaultChecked={desc} onChange={value => setDesc(value)} />
      </p>
      <AyForm
        ref={formRef}
        style={{ width: 600 }}
        desc={desc}
        readonly={readonly}
        span={24}
        fields={fields}
        onConfirm={handleConfirm}
      >
        <Space>
          <AyButton type="primary" htmlType="submit">
            提交
          </AyButton>
          <AyButton onClick={getFieldsValue}>获取当前输入值</AyButton>
          <AyButton onClick={setFieldsValue}>设置默认值</AyButton>
        </Space>
      </AyForm>
    </>
  )
}
```

## 参数详解

### group

如果是 group，一般中间会绘制一个分割线，如果不需要分割线，请使用 `input-group`。

```js
{
  title: '组合类型',
  // 此处的 type 为 group，表示子元素为组合类型
  type: 'group',
  // key 建议写一个
  key: '__group',
  //
  children: [
    {
      // 此处子元素写了 title 也不会展示，但是也可以写，因为还会影响到错误的校验信息和 placeholder
      title: '子元素A',
      key: 'a',
      required: true,
      span: 11
    },
    {
      // 此处的 key 随意
      key: '__divider',
      type: 'custom',
      // 此处 span 看着来
      span: 2,
      render: () => {
        // 你可以渲染一个分割线，此 class 会带一个默认的样式
        return <div className="ay-form-divider">-</div>
      }
    },
    {
      // 此处子元素写了 title 也不会展示，但是也可以写，因为还会影响到错误的校验信息和 placeholder
      title: '子元素B',
      key: 'b',
      required: true,
      span: 11
    }
  ]
}
```

### input-group

如果中间不需要分割符，且紧密的连接在一起，请使用 `input-group`。

```js
{
  title: '组合类型',
  // 此处的 type 为 group，表示子元素为组合类型
  type: 'group',
  // key 建议写一个
  key: '__group',
  children: [
    {
      // 此处子元素写了 title 也不会展示，但是也可以写，因为还会影响到错误的校验信息和 placeholder
      title: '子元素A',
      key: 'a',
      required: true,
      // span 设置无效，实际场景请不要写，如需设置宽度，请写 style
      span: 11,
      // 需要手动设置输入框宽度，不然会撑满，此处设置 span 无效
      props: {
        style: {
          width: '50%'
        }
      }
    },
    {
      // 此处子元素写了 title 也不会展示，但是也可以写，因为还会影响到错误的校验信息和 placeholder
      title: '子元素B',
      key: 'b',
      required: true,
      // span 设置无效，实际场景请不要写，如需设置宽度，请写 style
      span: 11,
      // 需要手动设置输入框宽度，不然会撑满，此处设置 span 无效
      props: {
        style: {
          width: '50%'
        }
      }
    }
  ]
}
```
